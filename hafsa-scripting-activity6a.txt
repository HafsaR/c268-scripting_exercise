function checkEquality(){
        echo "Enter number a and b"
        read a b
        if [[ ( $a =~ ^[0-9]+$ && $b =~ ^[0-9]+$ ) ]]
        then
                echo "a: $a, b: $b"
		diff=$(( a - b ))
  		echo "Difference between two numbers is $diff"
	else
                echo "Please enter valid numbers!"
        fi
}

checkEquality
