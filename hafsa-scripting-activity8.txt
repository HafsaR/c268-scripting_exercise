#!/bin/bash
: '
Author: Hafsa Degmaster
Version: 1.0
Description:
1. Prompts the user for a series of 5-10 integers.
- The user must be prompted for a minimum of 5 numbers.
- After the user has entered 5 numbers, the program should allow the user to stop entry or enter another number.
2. When the user chooses to stop or after 10 numbers have been entered, the program stops prompting for values and performs the following calculations:
- the product of the integers
- the average of the integers
- the sum of the integers
- the min of the integers
- the max of the integers
3. After performing the calculations, the program should display the following to the user:
- the values the user entered
- each of the calculations, using a phrase that identifies the value
'
arrayNumber=()

for i in {1..5}; 
do
	echo "Enter number"
	read number
	if ! [[ $number =~ ^[0-9]+$ ]]
	then
		exec >&2; echo "error: Not a number"; 
		exit 1
	else
		arrayNumber+=($number)
	fi	
done

# echo "${arrayNumber[@]}"    display array

arrToString="${arrayNumber[*]}"  # convert array to string

arrLength=${#arrayNumber[@]}     # count array length

: ' Initialize a variable '
product=1
average=0
sum=0
min=${arrayNumber[0]}
max=0
n=1


while [ $n -ne 6 ]
do

	echo "Choiose option:"
	echo "1. Product of the integers"\n
	echo "2. Average of the integers"\n
	echo "3. Sum of the integers"\n
	echo "4. Min of the integers"\n
	echo "5. Max of the integers"\n
	echo "6. exit"
	read n

	case $n in 
		1)
			for i in ${arrayNumber[@]}
			do
				product=$((product*i))
			done
			echo "Number the user entered: $arrToString"
			echo "The product of the integers: $product"
			;;
		2)
			for i in ${arrayNumber[@]}
			do
				average=$((average+i))
			done
			average=$(( average/arrLength ))
			echo "Number the user entered: $arrToString"
			echo "The average of the integers: $average"
			;;
		3)
			for i in ${arrayNumber[@]}
			do
				sum=$((sum+i))
			done
			echo "Number the user entered: $arrToString"
			echo "The sum of the integers: $sum"
			;;

		4)
			for i in ${arrayNumber[@]}
			do
				if [ $i -lt $min ]
				then
					min=$i
				fi
			done
			echo "Number the user entered: $arrToString"
			echo "The min of the integers: $min"
			;;
		5)

			for i in ${arrayNumber[@]}
			do
				if [ $i -gt $max ]
				then
					max=$i
				fi
			done
			echo "Number the user entered: $arrToString"
			echo "The max of the integers: $max"
			;;
		6)
			exit 0
			;;
		*)
			echo "Please choose valid option"
			;;
	esac

done
done
